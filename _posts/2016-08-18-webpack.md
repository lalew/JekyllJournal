Enable watch mode for Webpack via "build" and "start". Default file looked for is "webpack.config.js"
{% highlight shell %}
npm install --save-dev-server

  "scripts": {
    "build": "webpack --optimize-minimize",
    "start": "webpack-dev-server"
  },

{% endhighlight %}
